ALTER SESSION SET "_ORACLE_SCRIPT"=TRUE;
CREATE USER SISG IDENTIFIED BY UMG2024;
GRANT CONNECT, RESOURCE TO SISG;
GRANT UNLIMITED TABLESPACE TO SISG;

USE SISG; 

CREATE TABLE empleados (
    id NUMBER GENERATED BY DEFAULT ON NULL AS IDENTITY PRIMARY KEY,
    dpi CHAR(13) NOT NULL,
    nombre CHAR(50) NOT NULL,
    edad NUMBER NOT NULL,
    sexo CHAR(1) NOT NULL,
    email CHAR(50) NOT NULL,
    activo number NOT NULL	
);



CREATE TABLE EMPLEADOS 
(
  ID_EMPLEADO NUMBER(4) GENERATED ALWAYS AS IDENTITY NOT NULL 
, DPI CHAR(13) NOT NULL 
, NOMBRE CHAR(50) NOT NULL 
, EDAD NUMBER(2) NOT NULL
, SEXO CHAR(1) NOT NULL 
, EMAIL CHAR(50) NOT NULL 
, ACTIVO NUMBER(1) NOT NULL 
, CONSTRAINT EMPLEADOS_PK PRIMARY KEY 
  (
    ID_EMPLEADO 
  )
  ENABLE 
);
